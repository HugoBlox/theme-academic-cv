% adapted from psychtoolbox tutorials https://www.peterscarfe.com/ptbtutorials.html
% Read the image file
img = imread('rect.png');

% Specify the rect
rect = [0,0,100,100];

% Here we call some default settings for setting up Psychtoolbox
PsychDefaultSetup(2);

% Get the screen numbers
screens = Screen('Screens');

% Draw to the external screen if avaliable
screenNumber = max(screens);

% Open an on screen window
[window, windowRect] = PsychImaging('OpenWindow', screenNumber);

% Make the image into a texture
imgTexture = Screen('MakeTexture', window, img);

% Draw the image to rect
Screen('DrawTexture', window, imgTexture, [], rect, 0);

% Flip to the screen
Screen('Flip', window);

% Wait for two seconds
WaitSecs(2);

% Close the window
sca;
